pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/frisonexDev/JenkinsHL'
            }
        }
        stage('Restore') {
            steps {
                script {
                    // Restaura las dependencias
                    sh 'dotnet restore'
                }
            }
        }
        stage('Build') {
            steps {
                script {
                    // Compila el proyecto
                    sh 'dotnet build --configuration Release'
                }
            }
        }
        stage('Test') {
            steps {
                script {
                    // Ejecuta los tests
                    sh 'dotnet test --no-restore --verbosity normal'
                }
            }
        }
        stage('Publish') {
            steps {
                script {
                    // Publica el proyecto (esto puede variar según tu configuración)
                    sh 'dotnet publish --configuration Release --output ./publish'
                }
            }
        }
        stage('Deploy') {
            steps {
                script {
                    // Despliega la aplicación
                    // Aquí puedes agregar comandos para copiar archivos a un servidor, ejecutar scripts, etc.
                    // Por ejemplo:
                    sh 'scp -r ./publish usuario@servidor:/ruta/de/despliegue'
                }
            }
        }
    }
    post {
        success {
            echo 'El pipeline se ejecutó con éxito!'
        }
        failure {
            echo 'El pipeline falló.'
        }
    }
}
